# 表单高阶模版

## Usage

```js
import { FormRender } from '@deer-ui/admin-scaffold/template-engine'

class DemoPage extends React.Component {
  formOptions = [{
    type: 'input',
    ref: 'input',
    title: '实例输入框'
  }]
  state = {
    loading: false,
  },
  formBtns = [{
    text: '提交',
    action: () => {}
  }]
}

const FormTMPL = FormRender(DemoPage);

<FormTMPL />
```
